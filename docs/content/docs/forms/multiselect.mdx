## Usage

    <Tabs items={["preview" , "code"]}>
        <Tab value="preview">
            <div className="py-16 flex items-center justify-center">
                <MultiselectDemo />
            </div>
        </Tab>
        <Tab value="code">
            ```tsx
                import { Multiselect } from "@holo/forms";

                export const MultiselectDemo = () => {
                  const options = [
                    {
                      id: "1a",
                      label: "apple",
                      value: "apple",
                    },
                    {
                      id: "2a",
                      label: "cherry",
                      value: "cherry",
                    },
                    {
                      id: "3a",
                      label: "orange",
                      value: "orange",
                    },
                  ];
                  const items = [
                    {
                      id: "1a",
                      label: "apple",
                      value: "apple",
                    },
                    {
                      id: "2a",
                      label: "cherry",
                      value: "cherry",
                    },
                    {
                      id: "3a",
                      label: "orange",
                      value: "orange",
                    },
                  ];

                  return <Multiselect options={options} initialItems={items} />;
                };
            ```
        </Tab>
    </Tabs>
