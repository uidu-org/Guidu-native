    Card, CardContent,
    Carousel,
    CarouselContent,
    CarouselItem,
    CarouselNext,
    CarouselPrevious,
  } from "@uidu/core-ui"

  export default function CarouselDemo() {
    return (
      <Carousel className="w-full max-w-xs">

                </Card>
              </div>
            </CarouselItem>
          ))}
        </CarouselContent>
        <CarouselPrevious />

    )
  }
