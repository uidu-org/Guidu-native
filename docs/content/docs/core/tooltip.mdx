## Usage

    <Tabs items={["preview" , "code"]}>
        <Tab value="preview">
            <div className="py-16 flex items-center justify-center">
                <TooltipDemo />
            </div>
        </Tab>
        <Tab value="code">
            ```tsx
                "use client";

                import {
                Button,
                TooltipContent,
                TooltipProvider,
                TooltipRoot,
                TooltipTrigger,
                } from "@uidu/core-ui";

                export default function TooltipDemo() {
                return (
                    <TooltipProvider>
                    <TooltipRoot>
                        <TooltipTrigger asChild>
                        <Button variant="outline">Hover</Button>
                        </TooltipTrigger>
                        <TooltipContent>
                        <p>Add to library</p>
                        </TooltipContent>
                    </TooltipRoot>
                    </TooltipProvider>
                );
                }
            ```
        </Tab>
    </Tabs>
